async function loadMenu() { const response = await fetch("get_menu.php"); const items = await response.json(); const menuContainer = document.querySelector(".menu-items"); if (!menuContainer) return; menuContainer.innerHTML = ""; items.forEach(item => { menuContainer.innerHTML += `

${item.name}

${item.description}

Price: ₹${item.price} | Stock: ${item.stock}

`; }); } loadMenu();
